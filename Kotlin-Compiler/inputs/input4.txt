fun main() {
    println("Test 1: 1..5");
    for (i in 1..5) {
        println(i);
    }
    
    println("Test 2: 1..10 step 2");
    for (i in 1..10 step 2) {
        println(i);
    }
    
    println("Test 3: 5 downTo 1");
    for (i in 5 downTo 1) {
        println(i);
    }
    
    println("Test 4: 10 downTo 1 step 2");
    for (i in 10 downTo 1 step 2) {
        println(i);
    }
}
